[IMPROVE-4] thermal_ai.sh — GPU zone weight should scale with actual GPU load
FILE: scripts/thermal_ai.sh
FUNCTION: get_composite_temp()

PROBLEM:
gpuss-* zones always have weight 9 (just below cpuss). During
screen-off or light CPU work with idle GPU, gpuss readings still
drive the composite temp. This inflates the AI score unnecessarily
when the GPU is not contributing heat.

FIX:
In get_composite_temp(), read current GPU load once and dynamically
scale the gpuss zone weight before computing the weighted average:

get_composite_temp() {
    local tw=0 wt=0
    local gpu_load
    gpu_load=$(get_gpu_load)
    for entry in $ACTIVE_ZONES; do
        local zid="${entry%%:*}" w="${entry##*:}"
        local ztype
        ztype=$(cat "/sys/class/thermal/thermal_zone${zid}/type" 2>/dev/null)
        # Scale GPU zone weight by actual GPU activity
        case "$ztype" in
            gpuss-*)
                w=$((w * gpu_load / 100))
                [ "$w" -lt 1 ] && w=1  # never fully zero
                ;;
        esac
        local t
        t=$(cat "/sys/class/thermal/thermal_zone${zid}/temp" 2>/dev/null) || continue
        [ "$t" -gt 1000 ] 2>/dev/null && t=$((t/1000))
        [ "$t" -lt 0 ] 2>/dev/null && continue
        [ "$t" -gt 120 ] 2>/dev/null && continue
        tw=$((tw + t*w)); wt=$((wt + w))
    done
    if [ "$wt" -eq 0 ]; then
        TEMP_READ_FAILED=true; echo "45"; return
    fi
    TEMP_READ_FAILED=false
    echo $((tw/wt))
}

[FEATURE-3] Frame Stutter / Jank Detection via schedstat
FILE: scripts/advanced_ai.sh (add new function)
FILE: scripts/thermal_ai.sh (call from main_loop)

DESCRIPTION:
Read /proc/<game_pid>/schedstat to detect CPU scheduling delays on
the game's main thread. High wait time = threads queued but not
scheduled = visible jank. If stutter is detected while the policy
is "balanced", temporarily boost to "conservative" gaming mode
to give BIG/PRIME exclusive access to the game threads.

IMPLEMENTATION:
Add to scripts/advanced_ai.sh:

detect_frame_stutter() {
    local game_pkg="$1"
    [ -z "$game_pkg" ] && echo "false" && return

    # Find PID of the game process
    local game_pid=""
    for pid_cmdline in /proc/[0-9]*/cmdline; do
        local pid="${pid_cmdline%/cmdline}"
        pid="${pid##*/proc/}"
        local pkg
        pkg=$(cat "$pid_cmdline" 2>/dev/null | tr '\0' '\n' | head -1)
        [ "$pkg" = "$game_pkg" ] && game_pid="$pid" && break
    done
    [ -z "$game_pid" ] && echo "false" && return

    # schedstat field 2 = total wait time in nanoseconds
    local wait_ns
    wait_ns=$(awk '{print $2}' /proc/$game_pid/schedstat 2>/dev/null || echo "0")

    # Threshold: >50ms accumulated wait this tick = scheduling starvation
    if [ "$wait_ns" -gt 50000000 ]; then
        log_debug "Stutter detected: game=$game_pkg wait=${wait_ns}ns"
        echo "true"
        return
    fi
    echo "false"
}

Add to main_loop() in thermal_ai.sh, after new_policy is set
and after the cool-down override block:

        # Stutter override: jank during balanced -> force conservative gaming
        if $gaming && [ "$new_policy" = "balanced" ]; then
            local game_pkg
            game_pkg=$(get_current_game)
            local stutter
            stutter=$(detect_frame_stutter "$game_pkg")
            if $stutter; then
                log_info "Stutter override: balanced -> conservative (jank detected)"
                new_policy="conservative"
            fi
        fi

    echo "── ThermalAI Session History (last $n policy changes) ──"
    printf "  %-19s  %-6s  %-4s  %-12s  %s\n" "Time" "Temp" "GPU" "Gaming" "Policy"
    echo "  ─────────────────────────────────────────────────────────────"
    grep "Policy change:" "$LOG_FILE" 2>/dev/null | tail -n "$n" | while IFS= read -r line; do
        local ts cur gpu gaming policy
        ts=$(echo "$line"   | grep -o '\[.*\]' | head -1 | tr -d '[]')
        cur=$(echo "$line"  | grep -o 'temp=[0-9]*'    | cut -d= -f2)
        gpu=$(echo "$line"  | grep -o 'gpu=[0-9]*'     | cut -d= -f2)
        gaming=$(echo "$line" | grep -o 'gaming=[a-z]*' | cut -d= -f2)
        policy=$(echo "$line" | grep -o '-> .*'         | sed 's/-> //')
        printf "  %-19s  %-6s  %-4s  %-12s  %s\n" "$ts" "${cur}C" "${gpu}%" "$gaming" "$policy"
    done
    echo ""
    echo "  Tip: run 'thermalair logs' for full verbose output."
}

FILE: scripts/thermal_ai.sh
FUNCTION: main_loop()

DESCRIPTION:
Allow forcing a policy remotely from Tasker, ADB, shell scripts,
or automated test harnesses without needing the CLI binary in PATH:
  adb shell setprop thermalai.force_policy conservative
The daemon reads this Android system property on each poll cycle,
applies the policy if set, then clears it so it only fires once.

IMPLEMENTATION:
At the very top of the while loop in main_loop(), before the temp
read, add:

        # Property-based override (Tasker / ADB / automation)
        local forced_policy
        forced_policy=$(getprop thermalai.force_policy 2>/dev/null)
        if [ -n "$forced_policy" ] && [ "$forced_policy" != "none" ]; then
            case "$forced_policy" in
                performance|balanced|conservative|powersave|emergency_cool)
                    log_info "Property override: thermalai.force_policy=$forced_policy"
                    apply_thermal_policy "$forced_policy" "$gaming" "$temp"
                    CURRENT_POLICY="$forced_policy"
                    LAST_POLICY_CHANGE=$(date +%s)
                    setprop thermalai.force_policy "none" 2>/dev/null
                    ;;
                *)
                    log_warn "Invalid thermalai.force_policy value: $forced_policy"
                    setprop thermalai.force_policy "none" 2>/dev/null
                    ;;
            esac
        fi

[FEATURE-7] thermalair calibrate — CLI command for calibration management
FILE: system/bin/thermalair

DESCRIPTION:
Expose the persistent calibration system (added in BUG-8) through
a CLI subcommand. Allows users to check current offset and reset
calibration data without editing files manually.

IMPLEMENTATION:
Add new function to system/bin/thermalair:

cmd_calibrate() {
    local subcmd="${1:-status}"
    case "$subcmd" in
        reset)
            rm -f /data/local/tmp/thermalai.calibration
            rm -f /data/local/tmp/thermalai.calibration_offset
            echo "Calibration data cleared."
            echo "Thresholds will reset to base values on next daemon start."
            ;;
        status)
            local offset
            offset=$(cat /data/local/tmp/thermalai.calibration_offset 2>/dev/null || echo "0")
            local counter
            counter=$(cat /data/local/tmp/thermalai.calibration 2>/dev/null || echo "0")
            echo "Calibration offset : ${offset} degrees C (max allowed: -6)"
            echo "Hot-tick counter   : $counter / 60"
            ;;
        *)
            echo "Usage: thermalair calibrate [status|reset]"
            ;;
    esac
}

Add to the dispatch case block:
    calibrate) cmd_calibrate "$2" ;;
